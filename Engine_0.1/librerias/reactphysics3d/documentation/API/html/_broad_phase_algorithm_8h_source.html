225 // Method used to compare two pairs for sorting algorithm
226 inline bool BroadPhasePair::smallerThan(const BroadPhasePair& pair1, const BroadPhasePair& pair2) {
227 
228  if (pair1.collisionShape1ID < pair2.collisionShape1ID) return true;
229  if (pair1.collisionShape1ID == pair2.collisionShape1ID) {
230  return pair1.collisionShape2ID < pair2.collisionShape2ID;
231  }
232  return false;
233 }
