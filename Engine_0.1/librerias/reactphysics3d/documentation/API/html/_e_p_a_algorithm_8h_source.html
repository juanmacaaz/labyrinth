135 // Add a triangle face in the candidate triangle heap in the EPA algorithm
136 inline void EPAAlgorithm::addFaceCandidate(TriangleEPA* triangle, TriangleEPA** heap,
137  uint& nbTriangles, decimal upperBoundSquarePenDepth) {
138 
139  // If the closest point of the affine hull of triangle
140  // points is internal to the triangle and if the distance
141  // of the closest point from the origin is at most the
142  // penetration depth upper bound
143  if (triangle->isClosestPointInternalToTriangle() &&
144  triangle->getDistSquare() <= upperBoundSquarePenDepth) {
145 
146  // Add the triangle face to the list of candidates
147  heap[nbTriangles] = triangle;
148  nbTriangles++;
149  std::push_heap(&heap[0], &heap[nbTriangles], mTriangleComparison);
150  }
151 }
