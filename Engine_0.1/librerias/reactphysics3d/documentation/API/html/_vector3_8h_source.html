219 // Cross product of two vectors (inline)
220 inline Vector3 Vector3::cross(const Vector3& vector) const {
221  return Vector3(y * vector.z - z * vector.y,
222  z * vector.x - x * vector.z,
223  x * vector.y - y * vector.x);
224 }

226 // Normalize the vector
227 inline void Vector3::normalize() {
228  decimal l = length();
229  if (l < MACHINE_EPSILON) {
230  return;
231  }
232  x /= l;
233  y /= l;
234  z /= l;
235 }
