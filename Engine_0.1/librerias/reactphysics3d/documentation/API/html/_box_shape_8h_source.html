180 inline Vector3 BoxShape::getLocalSupportPointWithoutMargin(const Vector3& direction) const {

182  return Vector3(direction.x < decimal(0.0) ? -mHalfExtents.x : mHalfExtents.x,

187 // Return true if a point is inside the collision shape
188 inline bool BoxShape::testPointInside(const Vector3& localPoint, Collider* collider) const {
189  return (localPoint.x < mHalfExtents[0] && localPoint.x > -mHalfExtents[0] &&
190  localPoint.y < mHalfExtents[1] && localPoint.y > -mHalfExtents[1] &&
191  localPoint.z < mHalfExtents[2] && localPoint.z > -mHalfExtents[2]);
192 }

216 // Return the position of a given vertex
217 inline Vector3 BoxShape::getVertexPosition(uint vertexIndex) const {
218  assert(vertexIndex < getNbVertices());
219 
220  Vector3 extent = getHalfExtents();
221 
222  switch(vertexIndex) {
223  case 0: return Vector3(-extent.x, -extent.y, extent.z);
224  case 1: return Vector3(extent.x, -extent.y, extent.z);
225  case 2: return Vector3(extent.x, extent.y, extent.z);
226  case 3: return Vector3(-extent.x, extent.y, extent.z);
227  case 4: return Vector3(-extent.x, -extent.y, -extent.z);
228  case 5: return Vector3(extent.x, -extent.y, -extent.z);
229  case 6: return Vector3(extent.x, extent.y, -extent.z);
230  case 7: return Vector3(-extent.x, extent.y, -extent.z);
231  }
232 
233  assert(false);
234  return Vector3::zero();
235 }

237 // Return the normal vector of a given face of the polyhedron
238 inline Vector3 BoxShape::getFaceNormal(uint faceIndex) const {
239  assert(faceIndex < getNbFaces());
240 
241  switch(faceIndex) {
242  case 0: return Vector3(0, 0, 1);
243  case 1: return Vector3(1, 0, 0);
244  case 2: return Vector3(0, 0, -1);
245  case 3: return Vector3(-1, 0, 0);
246  case 4: return Vector3(0, -1, 0);
247  case 5: return Vector3(0, 1, 0);
248  }
249 
250  assert(false);
251  return Vector3::zero();
252 }
