387 // Add a body to the collision detection
388 inline void CollisionDetectionSystem::addCollider(Collider* collider, const AABB& aabb) {
389 
390  // Add the body to the broad-phase
391  mBroadPhaseSystem.addCollider(collider, aabb);
392 
393  int broadPhaseId = mCollidersComponents.getBroadPhaseId(collider->getEntity());
394 
395  assert(!mMapBroadPhaseIdToColliderEntity.containsKey(broadPhaseId));
396 
397  // Add the mapping between the collider broad-phase id and its entity
398  mMapBroadPhaseIdToColliderEntity.add(Pair<int, Entity>(broadPhaseId, collider->getEntity()));
399 }
