319 // Set whether a debug info is displayed or not
324 inline void DebugRenderer::setIsDebugItemDisplayed(DebugItem item, bool isDisplayed) {
325  const uint32 itemFlag = static_cast<uint32>(item);
326  uint32 resetBit = ~(itemFlag);
327  mDisplayedDebugItems &= resetBit;
328  if (isDisplayed) {
329  mDisplayedDebugItems |= itemFlag;
330  }
