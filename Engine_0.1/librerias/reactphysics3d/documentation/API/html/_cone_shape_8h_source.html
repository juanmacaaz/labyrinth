162 // Return the local inertia tensor of the collision shape
168 inline void ConeShape::computeLocalInertiaTensor(Matrix3x3& tensor, decimal mass) const {
169  decimal rSquare = mRadius * mRadius;
170  decimal diagXZ = decimal(0.15) * mass * (rSquare + mHalfHeight);
171  tensor.setAllValues(diagXZ, 0.0, 0.0,
172  0.0, decimal(0.3) * mass * rSquare,
173  0.0, 0.0, 0.0, diagXZ);
174 }

176 // Return true if a point is inside the collision shape
177 inline bool ConeShape::testPointInside(const Vector3& localPoint, ProxyShape* proxyShape) const {
178  const decimal radiusHeight = mRadius * (-localPoint.y + mHalfHeight) /
179  (mHalfHeight * decimal(2.0));
180  return (localPoint.y < mHalfHeight && localPoint.y > -mHalfHeight) &&
181  (localPoint.x * localPoint.x + localPoint.z * localPoint.z < radiusHeight *radiusHeight);
