165 // Return true if the current AABB is overlapping with the AABB in argument.
167 inline bool AABB::testCollision(const AABB& aabb) const {
168  if (mMaxCoordinates.x < aabb.mMinCoordinates.x ||
169  aabb.mMaxCoordinates.x < mMinCoordinates.x) return false;
170  if (mMaxCoordinates.y < aabb.mMinCoordinates.y ||
171  aabb.mMaxCoordinates.y < mMinCoordinates.y) return false;
172  if (mMaxCoordinates.z < aabb.mMinCoordinates.z||
173  aabb.mMaxCoordinates.z < mMinCoordinates.z) return false;
174  return true;
175 }
176 
177 // Return the volume of the AABB
178 inline decimal AABB::getVolume() const {
179  const Vector3 diff = mMaxCoordinates - mMinCoordinates;
180  return (diff.x * diff.y * diff.z);
181 }

183 // Return true if the AABB of a triangle intersects the AABB
184 inline bool AABB::testCollisionTriangleAABB(const Vector3* trianglePoints) const {
185 
186  if (min3(trianglePoints[0].x, trianglePoints[1].x, trianglePoints[2].x) > mMaxCoordinates.x) return false;
187  if (min3(trianglePoints[0].y, trianglePoints[1].y, trianglePoints[2].y) > mMaxCoordinates.y) return false;
188  if (min3(trianglePoints[0].z, trianglePoints[1].z, trianglePoints[2].z) > mMaxCoordinates.z) return false;
189 
190  if (max3(trianglePoints[0].x, trianglePoints[1].x, trianglePoints[2].x) < mMinCoordinates.x) return false;
191  if (max3(trianglePoints[0].y, trianglePoints[1].y, trianglePoints[2].y) < mMinCoordinates.y) return false;
192  if (max3(trianglePoints[0].z, trianglePoints[1].z, trianglePoints[2].z) < mMinCoordinates.z) return false;
193 
194  return true;
195 }

197 // Return true if a point is inside the AABB
198 inline bool AABB::contains(const Vector3& point) const {
199 
200  return (point.x >= mMinCoordinates.x - MACHINE_EPSILON && point.x <= mMaxCoordinates.x + MACHINE_EPSILON &&
201  point.y >= mMinCoordinates.y - MACHINE_EPSILON && point.y <= mMaxCoordinates.y + MACHINE_EPSILON &&
202  point.z >= mMinCoordinates.z - MACHINE_EPSILON && point.z <= mMaxCoordinates.z + MACHINE_EPSILON);
203 }
