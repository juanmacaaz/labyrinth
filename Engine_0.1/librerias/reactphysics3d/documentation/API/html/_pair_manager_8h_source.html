217 // Return the next power of two of a 32bits integer using a SWAR algorithm
218 inline luint PairManager::computeNextPowerOfTwo(luint number) {
219  number |= (number >> 1);
220  number |= (number >> 2);
221  number |= (number >> 4);
222  number |= (number >> 8);
223  number |= (number >> 16);
224  return number+1;

249 // This method returns an hash value for a 32 bits key.
253 inline int PairManager::computeHash32Bits(int key) const {
254  key += ~(key << 15);
255  key ^= (key >> 10);
256  key += (key << 3);
257  key ^= (key >> 6);
258  key += ~(key << 11);
259  key ^= (key >> 16);
260  return key;
261 }
