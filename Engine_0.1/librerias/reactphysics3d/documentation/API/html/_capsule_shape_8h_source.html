203 // Return a local support point in a given direction without the object margin.
210 inline Vector3 CapsuleShape::getLocalSupportPointWithoutMargin(const Vector3& direction) const {
211 
212  // Support point top sphere
213  decimal dotProductTop = mHalfHeight * direction.y;
214 
215  // Support point bottom sphere
216  decimal dotProductBottom = -mHalfHeight * direction.y;
217 
218  // Return the point with the maximum dot product
219  if (dotProductTop > dotProductBottom) {
220  return Vector3(0, mHalfHeight, 0);
221  }
222  else {
223  return Vector3(0, -mHalfHeight, 0);
224  }
225 }
