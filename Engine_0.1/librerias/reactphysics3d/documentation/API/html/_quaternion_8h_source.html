356 // Overloaded operator for the multiplication of two quaternions
357 inline Quaternion Quaternion::operator*(const Quaternion& quaternion) const {
358 
359  /* The followin code is equivalent to this
360  return Quaternion(w * quaternion.w - getVectorV().dot(quaternion.getVectorV()),
361  w * quaternion.getVectorV() + quaternion.w * getVectorV() +
362  getVectorV().cross(quaternion.getVectorV()));
363  */
364 
365  return Quaternion(w * quaternion.x + quaternion.w * x + y * quaternion.z - z * quaternion.y,
366  w * quaternion.y + quaternion.w * y + z * quaternion.x - x * quaternion.z,
367  w * quaternion.z + quaternion.w * z + x * quaternion.y - y * quaternion.x,
368  w * quaternion.w - x * quaternion.x - y * quaternion.y - z * quaternion.z);
369 }

371 // Overloaded operator for the multiplication with a vector.
373 inline Vector3 Quaternion::operator*(const Vector3& point) const {
374 
375  /* The following code is equivalent to this
376  * Quaternion p(point.x, point.y, point.z, 0.0);
377  * return (((*this) * p) * getConjugate()).getVectorV();
378  */
379 
380  const decimal prodX = w * point.x + y * point.z - z * point.y;
381  const decimal prodY = w * point.y + z * point.x - x * point.z;
382  const decimal prodZ = w * point.z + x * point.y - y * point.x;
383  const decimal prodW = -x * point.x - y * point.y - z * point.z;
384  return Vector3(w * prodX - prodY * z + prodZ * y - prodW * x,
385  w * prodY - prodZ * x + prodX * z - prodW * y,
386  w * prodZ - prodX * y + prodY * x - prodW * z);
387 }
