83 inline void NoBroadPhaseAlgorithm::addObject(CollisionBody* body, const AABB& aabb) {
84 
85  // For each body that is already in the world
86  for (std::set<CollisionBody*>::iterator it = mBodies.begin(); it != mBodies.end(); ++it) {
87 
88  // Add an overlapping pair with the new body
89  mPairManager.addPair(*it, body);
90  }
91 
92  // Add the new body into the list of bodies
93  mBodies.insert(body);
94 }

96 // Notify the broad-phase about an object that has been removed from the world
97 inline void NoBroadPhaseAlgorithm::removeObject(CollisionBody* body) {
98 
99  // For each body that is in the world
100  for (std::set<CollisionBody*>::iterator it = mBodies.begin(); it != mBodies.end(); ++it) {
101 
102  if ((*it)->getID() != body->getID()) {
103 
104  // Remove the overlapping pair with the new body
105  mPairManager.removePair((*it)->getID(), body->getID());
106  }
107  }
108 
109  // Remove the body from the broad-phase
110  mBodies.erase(body);
111 }
