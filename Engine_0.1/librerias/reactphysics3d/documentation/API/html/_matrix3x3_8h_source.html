223 // Return the determinant of the matrix
224 inline decimal Matrix3x3::getDeterminant() const {
225 
226  // Compute and return the determinant of the matrix
227  return (mRows[0][0]*(mRows[1][1]*mRows[2][2]-mRows[2][1]*mRows[1][2]) -
228  mRows[0][1]*(mRows[1][0]*mRows[2][2]-mRows[2][0]*mRows[1][2]) +
229  mRows[0][2]*(mRows[1][0]*mRows[2][1]-mRows[2][0]*mRows[1][1]));
230 }

232 // Return the trace of the matrix

237 }
