361  mFileStream(filePath, std::ios::binary) {
362

378 
379  // Close the stream
380  mFileStream.close();
381  }
382  }

385  virtual void write(const time_t& time, const std::string& physicsWorldName, const std::string& message, Level level, Category category,
386  const char* filename, int lineNumber) override {
387 
388  if (static_cast<int>(level) <= static_cast<int>(maxLevelFlag)) {
389  mFileStream << formatter->format(time, physicsWorldName, message, level, category, filename, lineNumber) << std::endl << std::flush;
