181 
184 inline Vector3 ConvexMeshShape::getLocalInertiaTensor(decimal mass) const {
185  const decimal factor = (decimal(1.0) / decimal(3.0)) * mass;
186  const Vector3 realExtent = decimal(0.5) * (mMaxBounds - mMinBounds);
187  assert(realExtent.x > 0 && realExtent.y > 0 && realExtent.z > 0);
188  const decimal xSquare = realExtent.x * realExtent.x;
189  const decimal ySquare = realExtent.y * realExtent.y;
190  const decimal zSquare = realExtent.z * realExtent.z;
191  return Vector3(factor * (ySquare + zSquare), factor * (xSquare + zSquare), factor * (xSquare + ySquare));
192 }
