184 // Compute the point of an object closest to the origin
185 inline Vector3 TriangleEPA::computeClosestPointOfObject(const Vector3* supportPointsOfObject) const{
186  const Vector3& p0 = supportPointsOfObject[mIndicesVertices[0]];
187  return p0 + decimal(1.0)/mDet * (mLambda1 * (supportPointsOfObject[mIndicesVertices[1]] - p0) +
188  mLambda2 * (supportPointsOfObject[mIndicesVertices[2]] - p0));
189 }
