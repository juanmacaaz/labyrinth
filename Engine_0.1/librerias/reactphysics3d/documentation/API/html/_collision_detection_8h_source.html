290 // Return the corresponding narrow-phase algorithm
291 inline NarrowPhaseAlgorithm* CollisionDetection::selectNarrowPhaseAlgorithm(const CollisionShapeType& shape1Type,
292  const CollisionShapeType& shape2Type) const {
293 
294  uint shape1Index = static_cast<unsigned int>(shape1Type);
295  uint shape2Index = static_cast<unsigned int>(shape2Type);
296 
297  // Swap the shape types if necessary
298  if (shape1Index > shape2Index) {
299  const uint tempIndex = shape1Index;
300  shape1Index = shape2Index;
301  shape2Index = tempIndex;
302  }
303 
304  assert(shape1Index <= shape2Index);
305 
306  return mCollisionMatrix[shape1Index][shape2Index];
307 }
