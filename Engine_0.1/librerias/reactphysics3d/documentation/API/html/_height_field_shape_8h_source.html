189 // Return the height of a given (x,y) point in the height field
190 inline decimal HeightFieldShape::getHeightAt(int x, int y) const {
191 
192  assert(x >= 0 && x < mNbColumns);
193  assert(y >= 0 && y < mNbRows);
194 
195  switch(mHeightDataType) {
196  case HeightDataType::HEIGHT_FLOAT_TYPE : return ((float*)mHeightFieldData)[y * mNbColumns + x];
197  case HeightDataType::HEIGHT_DOUBLE_TYPE : return ((double*)mHeightFieldData)[y * mNbColumns + x];
198  case HeightDataType::HEIGHT_INT_TYPE : return ((int*)mHeightFieldData)[y * mNbColumns + x] * mIntegerHeightScale;
199  default: assert(false); return 0;
200  }
201 }

203 // Return the closest inside integer grid value of a given floating grid value
204 inline int HeightFieldShape::computeIntegerGridValue(decimal value) const {
205  return (value < decimal(0.0)) ? value - decimal(0.5) : value + decimal(0.5);
206 }
