182 // Return an interpolated transform
183 inline Transform Transform::interpolateTransforms(const Transform& oldTransform,
184  const Transform& newTransform,
185  decimal interpolationFactor) {
186 
187  Vector3 interPosition = oldTransform.mPosition * (decimal(1.0) - interpolationFactor) +
188  newTransform.mPosition * interpolationFactor;
189 
190  Quaternion interOrientation = Quaternion::slerp(oldTransform.mOrientation,
191  newTransform.mOrientation,
192  interpolationFactor);
193 
194  return Transform(interPosition, interOrientation);
195 }

212 // Operator of multiplication of a transform with another one
213 inline Transform Transform::operator*(const Transform& transform2) const {
214 
215  // The following code is equivalent to this
216  //return Transform(mPosition + mOrientation * transform2.mPosition,
217  // mOrientation * transform2.mOrientation);
218 
219  const decimal prodX = mOrientation.w * transform2.mPosition.x + mOrientation.y * transform2.mPosition.z
220  - mOrientation.z * transform2.mPosition.y;
221  const decimal prodY = mOrientation.w * transform2.mPosition.y + mOrientation.z * transform2.mPosition.x
222  - mOrientation.x * transform2.mPosition.z;
223  const decimal prodZ = mOrientation.w * transform2.mPosition.z + mOrientation.x * transform2.mPosition.y
224  - mOrientation.y * transform2.mPosition.x;
225  const decimal prodW = -mOrientation.x * transform2.mPosition.x - mOrientation.y * transform2.mPosition.y
226  - mOrientation.z * transform2.mPosition.z;
227 
228  return Transform(Vector3(mPosition.x + mOrientation.w * prodX - prodY * mOrientation.z + prodZ * mOrientation.y - prodW * mOrientation.x,
229  mPosition.y + mOrientation.w * prodY - prodZ * mOrientation.x + prodX * mOrientation.z - prodW * mOrientation.y,
230  mPosition.z + mOrientation.w * prodZ - prodX * mOrientation.y + prodY * mOrientation.x - prodW * mOrientation.z),
231  Quaternion(mOrientation.w * transform2.mOrientation.x + transform2.mOrientation.w * mOrientation.x
232  + mOrientation.y * transform2.mOrientation.z - mOrientation.z * transform2.mOrientation.y,
233  mOrientation.w * transform2.mOrientation.y + transform2.mOrientation.w * mOrientation.y
234  + mOrientation.z * transform2.mOrientation.x - mOrientation.x * transform2.mOrientation.z,
235  mOrientation.w * transform2.mOrientation.z + transform2.mOrientation.w * mOrientation.z
236  + mOrientation.x * transform2.mOrientation.y - mOrientation.y * transform2.mOrientation.x,
237  mOrientation.w * transform2.mOrientation.w - mOrientation.x * transform2.mOrientation.x
238  - mOrientation.y * transform2.mOrientation.y - mOrientation.z * transform2.mOrientation.z));
239 }
