62  void allocate(size_t capacity) {
63 
64  T* newArray = static_cast<T*>(mAllocator.allocate(capacity * sizeof(T)));
65  assert(newArray != nullptr);
66 
67  // If there
68  if (mCapacity > 0) {
69 
70  if (mNbElements > 0) {
71 
72  // Copy the previous items in the new array
73  std::uninitialized_copy(mArray, mArray + mNbElements, newArray);
74  }
75 
76  // Release memory of the previous array
77  mAllocator.release(mArray, mCapacity * sizeof(T));
78  }
79 
80  mArray = newArray;
81 
82  mCapacity = capacity;
83  }

99  Stack(const Stack& stack)
100  :mAllocator(stack.mAllocator), mArray(nullptr),
101  mNbElements(stack.mNbElements), mCapacity(stack.mCapacity) {
102 
103  if (mCapacity > 0) {
104 
105  // Allocate memory for the buckets
106  mArray = static_cast<T*>(mAllocator.allocate(mCapacity * sizeof(T)));
107  assert(mArray != nullptr);
108 
109  if (mNbElements > 0) {
110 
111  // Copy the items
112  std::uninitialized_copy(stack.mArray, stack.mArray + mNbElements, mArray);
113  }
114  }
115  }
116 
118  ~Stack() {
119 
120  clear();
121 
122  if (mCapacity > 0) {
123 
124  // Release the memory allocated on the heap
125  mAllocator.release(mArray, mCapacity * sizeof(T));
126  }
127  }
128 
130  void clear() {
131 
132  // Destruct the items
133  for (size_t i = 0; i < mNbElements; i++) {
134  mArray[i].~T();
135  }
136 
137  mNbElements = 0;
138  }
139 
141  void push(const T& element) {
142 
143  // If we need to allocate more elements
144  if (mNbElements == mCapacity) {
145 
146  allocate(mCapacity > 0 ? mCapacity * 2 : 1);
147  }
148 
149  // Copy the item into the array
150  new (mArray + mNbElements) T(element);
151 
152  mNbElements++;
153  }
154 
156  T pop() {
157 
158  assert(mNbElements > 0);
159 
160  mNbElements--;
161 
162  // Copy the item
163  T item = mArray[mNbElements];
164 
165  // Call the destructor
166  mArray[mNbElements].~T();
167 
168  return item;
169  }
170 
172  size_t size() const {
173  return mNbElements;
174  }
175 
177  size_t capacity() const {
178  return mCapacity;
179  }
180 };
