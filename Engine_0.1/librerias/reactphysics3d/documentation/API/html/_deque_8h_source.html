92  T& getItem(size_t virtualIndex) const {
93 
94  // If the virtual index is valid
95  if (virtualIndex < mSize) {
96 
97  size_t chunkIndex = mFirstChunkIndex;
98  size_t itemIndex = mFirstItemIndex;
99 
100  const size_t nbItemsFirstChunk = CHUNK_NB_ITEMS - mFirstItemIndex;
101  if (virtualIndex < nbItemsFirstChunk) {
102  itemIndex += virtualIndex;
103  }
104  else {
105 
106  virtualIndex -= nbItemsFirstChunk;
107  chunkIndex++;
108 
109  chunkIndex += virtualIndex / CHUNK_NB_ITEMS;
110  itemIndex = virtualIndex % CHUNK_NB_ITEMS;
111  }
112 
113  return mChunks[chunkIndex][itemIndex];
114  }
115  else {
116  assert(false);
117  }
118  }
119 
121  void expandChunks(size_t atLeastNbChunks = 0) {
122 
123  // If it is not necessary to expand the chunks
124  if (atLeastNbChunks > 0 && atLeastNbChunks <= mNbChunks) {
125  return;
126  }
127 
128  size_t newNbChunks = mNbChunks == 0 ? 3 : 2 * mNbChunks - 1;
129  if (atLeastNbChunks > 0 && newNbChunks < atLeastNbChunks) {
130  newNbChunks = size_t(atLeastNbChunks / 2) * 2 + 1;
131  }
132  const size_t halfNbChunksToAdd = mNbChunks == 0 ? 1 : (mNbChunks - 1) / 2;
133 
134  // Allocate memory for the new array of pointers to chunk
135  void* newMemory = mAllocator.allocate(newNbChunks * sizeof(T*));
136  assert(newMemory != nullptr);
137  T** newChunks = static_cast<T**>(newMemory);
138 
139  // If chunks have already been allocated
140  if (mNbChunks > 0) {
141 
142  // Copy the pointers to the previous chunks to the new allocated memory location
143  std::uninitialized_copy(mChunks, mChunks + mNbChunks, newChunks + halfNbChunksToAdd);
144 
145  // Release the previously allocated memory
146  mAllocator.release(mChunks, mNbChunks * sizeof(T*));
147  }
148 
149  mChunks = newChunks;
150 
151  // If we need to allocate the first chunk (in the middle of the chunks array)
152  if (mNbChunks == 0) {
153  mChunks[newNbChunks / 2] = static_cast<T*>(mAllocator.allocate(sizeof(T) * CHUNK_NB_ITEMS));
154  assert(mChunks[newNbChunks / 2] != nullptr);
155  }
156 
157  mNbChunks = newNbChunks;
158 
159  // Allocate memory for each new chunk
160  for (size_t i=0; i < halfNbChunksToAdd; i++) {
161 
162  // Allocate memory for the new chunk
163  mChunks[i] = static_cast<T*>(mAllocator.allocate(sizeof(T) * CHUNK_NB_ITEMS));
164  assert(mChunks[i] != nullptr);
165 
166  mChunks[mNbChunks - 1 - i] = static_cast<T*>(mAllocator.allocate(sizeof(T) * CHUNK_NB_ITEMS));
167  assert(mChunks[mNbChunks - 1 -i] != nullptr);
168  }
169 
170  // Update the first and last chunk index
171  mFirstChunkIndex += halfNbChunksToAdd;
172  mLastChunkIndex += halfNbChunksToAdd;
173  }

344  expandChunks(deque.mNbChunks);
345

347 
348  const size_t dequeHalfSize1 = std::ceil(deque.mSize / 2.0f);
349  const size_t dequeHalfSize2 = deque.mSize - dequeHalfSize1;

351  // Add the items into the deque
352  for(size_t i=0; i < dequeHalfSize1; i++) {
353  addFront(deque[dequeHalfSize1 - 1 - i]);
354  }

377  void addBack(const T& element) {
378 
379  // If we need to add the item in a another chunk
380  if (mLastItemIndex == CHUNK_NB_ITEMS - 1) {
381 
382  // If we need to add more chunks
383  if (mLastChunkIndex == mNbChunks - 1) {
384 
385  // Add more chunks
386  expandChunks();
387  }
388 
389  mLastItemIndex = 0;
390  mLastChunkIndex++;
391  }
392  else if (mSize != 0) {
393  mLastItemIndex++;
394  }
395 
396  // Construct the element at its location in the chunk
397  new (static_cast<void*>(&(mChunks[mLastChunkIndex][mLastItemIndex]))) T(element);
398 
399  mSize++;
400 
401  assert(mFirstChunkIndex >= 0 && mLastChunkIndex < mNbChunks);
402  assert(mFirstItemIndex >= 0 && mFirstItemIndex < CHUNK_NB_ITEMS);
403  assert(mLastItemIndex >= 0 && mLastItemIndex < CHUNK_NB_ITEMS);
404  assert(mFirstChunkIndex <= mLastChunkIndex);
405  }

408  void addFront(const T& element) {
409 
410  // If we need to add the item in another chunk
411  if (mFirstItemIndex == 0) {
412

415

419

429

439  void popFront() {
440 
441  if (mSize > 0) {
442 
443  // Call the destructor of the first element
444  mChunks[mFirstChunkIndex][mFirstItemIndex].~T();
445 
446  mSize--;
447 
448  if (mSize == 0) {
449  mFirstChunkIndex = mNbChunks / 2;
450  mFirstItemIndex = CHUNK_FIRST_ITEM_INDEX;
451  mLastChunkIndex = mFirstChunkIndex;
452  mLastItemIndex = CHUNK_FIRST_ITEM_INDEX;
453  }
454  else if (mFirstItemIndex == CHUNK_NB_ITEMS - 1){
455  mFirstChunkIndex++;
456  mFirstItemIndex = 0;
457  }
458  else {
459  mFirstItemIndex++;
460  }
461 
462  assert(mFirstChunkIndex >= 0 && mLastChunkIndex < mNbChunks);
463  assert(mFirstItemIndex >= 0 && mFirstItemIndex < CHUNK_NB_ITEMS);
464  assert(mLastItemIndex >= 0 && mLastItemIndex < CHUNK_NB_ITEMS);
465  assert(mFirstChunkIndex <= mLastChunkIndex);
466  }
467  }

477  mSize--;
