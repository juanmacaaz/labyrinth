159 // Return the local inertia tensor of the cylinder
165 inline void CylinderShape::computeLocalInertiaTensor(Matrix3x3& tensor, decimal mass) const {
166  decimal height = decimal(2.0) * mHalfHeight;
167  decimal diag = (decimal(1.0) / decimal(12.0)) * mass * (3 * mRadius * mRadius + height * height);
168  tensor.setAllValues(diag, 0.0, 0.0, 0.0,
169  decimal(0.5) * mass * mRadius * mRadius, 0.0,
170  0.0, 0.0, diag);
171 }
