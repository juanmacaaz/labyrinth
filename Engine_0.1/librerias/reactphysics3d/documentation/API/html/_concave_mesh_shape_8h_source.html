234 // Called when a overlapping node has been found during the call to
235 // DynamicAABBTree:reportAllShapesOverlappingWithAABB()
236 inline void ConvexTriangleAABBOverlapCallback::notifyOverlappingNode(int nodeId) {
237 
238  // Get the node data (triangle index and mesh subpart index)
239  int32* data = mDynamicAABBTree.getNodeDataInt(nodeId);
240 
241  // Get the triangle vertices for this node from the concave mesh shape
242  Vector3 trianglePoints[3];
243  mConcaveMeshShape.getTriangleVertices(data[0], data[1], trianglePoints);
244 
245  // Get the vertices normals of the triangle
246  Vector3 verticesNormals[3];
247  mConcaveMeshShape.getTriangleVerticesNormals(data[0], data[1], verticesNormals);
248 
249  // Call the callback to test narrow-phase collision with this triangle
250  mTriangleTestCallback.testTriangle(trianglePoints, verticesNormals, mConcaveMeshShape.computeTriangleShapeId(data[0], data[1]));
251 }
