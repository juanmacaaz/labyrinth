160 inline void SphereShape::getLocalBounds(Vector3& min, Vector3& max) const {
161 
162  // Maximum bounds

173 // Return the local inertia tensor of the sphere
177 inline Vector3 SphereShape::getLocalInertiaTensor(decimal mass) const {
178  decimal diag = decimal(0.4) * mass * mMargin * mMargin;
179  return Vector3(diag, diag, diag);
180 }
181 
182 // Compute and return the volume of the collision shape
183 inline decimal SphereShape::getVolume() const {
184  return decimal(4.0) / decimal(3.0) * reactphysics3d::PI * mMargin * mMargin * mMargin;
185 }

187 // Return true if a point is inside the collision shape
188 inline bool SphereShape::testPointInside(const Vector3& localPoint, Collider* collider) const {
189  return (localPoint.lengthSquare() < mMargin * mMargin);
190 }
