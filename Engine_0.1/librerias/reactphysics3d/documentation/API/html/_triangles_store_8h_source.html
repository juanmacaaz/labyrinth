116 inline TriangleEPA* TrianglesStore::newTriangle(const Vector3* vertices,
117  uint v0,uint v1, uint v2) {
118  TriangleEPA* newTriangle = NULL;
119 
120  // If we have not reached the maximum number of triangles
121  if (mNbTriangles != MAX_TRIANGLES) {
122  newTriangle = &mTriangles[mNbTriangles++];
123  new (newTriangle) TriangleEPA(v0, v1, v2);
124  if (!newTriangle->computeClosestPoint(vertices)) {
125  mNbTriangles--;
126  newTriangle = NULL;
127  }
128  }
129 
130  // Return the new triangle
131  return newTriangle;
132 }
