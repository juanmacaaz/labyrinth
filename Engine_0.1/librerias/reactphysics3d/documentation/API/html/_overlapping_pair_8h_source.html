252 // Return the pair of bodies index
253 inline OverlappingPair::OverlappingPairId OverlappingPair::computeID(ProxyShape* shape1, ProxyShape* shape2) {
254  assert(shape1->getBroadPhaseId() >= 0 && shape2->getBroadPhaseId() >= 0);
255 
256  // Construct the pair of body index
257  OverlappingPairId pairID = shape1->getBroadPhaseId() < shape2->getBroadPhaseId() ?
258  OverlappingPairId(shape1->getBroadPhaseId(), shape2->getBroadPhaseId()) :
259  OverlappingPairId(shape2->getBroadPhaseId(), shape1->getBroadPhaseId());
260  assert(pairID.first != pairID.second);
261  return pairID;
262 }
